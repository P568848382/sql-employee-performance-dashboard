---- Pay zone + salary sanity check

select payzone,
	   count(*) as headcount,
	   round(min(monthly_salary),0) as min_salary,
	   round(max(monthly_salary),0)  as max_salary,
	   round(avg(monthly_salary),0) as avg_salary
from employee_performance
group by payzone
order by payzone;
o/p:-
"payzone"	"headcount"	"min_salary"	"max_salary"	"avg_salary"
"Zone A"	1062	33850	51250	42954
"Zone B"	985	60400	84700	73014
"Zone C"	953	97850	132250	115741
